// How to colour the -top texture
const colour = new Color(1, 1, 1, 0.5);

const clear = extend(Router, "clear-router", {
	load() {
		this.super$load();
		this.bottomRegion = Core.atlas.find("clear-router-bottom");
		this.topRegion = Core.atlas.find("clear-router-top");
	},

	drawBase(tile){
		const building = tile.build;
		Draw.rect(this.bottomRegion, tile.drawx(), tile.drawy());
		Draw.color(colour);
		Draw.rect(this.topRegion, tile.drawx(), tile.drawy());
		Draw.color();
		if (building.items.total() != 0) {
			Draw.rect(building.items.first().fullIcon, tile.drawx(), tile.drawy());
		}
	}
});

module.exports = clear;
